#   defining functions that will be used in later computations   #

def symmetrises(A, diag=False):
    if diag:
        return np.tril(A, 0) + np.transpose(np.tril(A, -1))
    else:
        return np.tril(A, -1) + np.transpose(np.tril(A, -1))


def gaussian_ellipse(mean, cov):
    if mean.shape == (1, 2):
        mean = np.array(mean)[0]
    v, w = np.linalg.eigh(cov)
    u = w[0] / np.linalg.norm(w[0])
    rtheta = np.arctan2(u[1], u[0])
    v = 3. * np.sqrt(2.) * np.sqrt(v)
    width = v[0]; height = v[1]

    R = np.array([
        [np.cos(rtheta), -np.sin(rtheta)],
        [np.sin(rtheta), np.cos(rtheta)],
    ])
    theta = np.deg2rad(np.arange(0.0, 360.0, 1.0))
    x = 0.5 * width * np.cos(theta)
    y = 0.5 * height * np.sin(theta)
    x, y = np.dot(R, np.array([x, y]))
    x += mean[0]
    y += mean[1]

    return [x, y]


def right_embed(As, K):

    T, n, _ = As.shape
    A = As.reshape(T * n, n)

    U, S, Vt = np.linalg.svd(A)
    VA = Vt.T

    YA = VA[:, :K] * np.sqrt(S[:K])
    return YA.reshape(T, n, K)


##########################################################################
#   Plotting function for different node counts at the first time step   #
##########################################################################

def plot_time_step_with_nodes(B, K, node_counts):
    fig, axs = plt.subplots(1, len(node_counts), figsize=(15, 5), sharex=True, sharey=True)

    for i, n in enumerate(node_counts):
        np.random.seed(0)
        Z = np.repeat(range(K), n // K)  # Community labels for each node
        colours = np.array(list(mpl.colors.TABLEAU_COLORS.keys())[:K])
        Zcol = colours[Z]  # Assign colors based on community labels

        # Generate adjacency matrix for the first time step
        A = symmetrises(stats.bernoulli.rvs(B[Z, :][:, Z]))

        # Perform right embedding
        As = np.array([A])
        YA_UASE = right_embed(As, K)

        # Scatter plot of the embedding for this node count
        axs[i].scatter(YA_UASE[0, :, 0], YA_UASE[0, :, 1], marker='.', s=5, c=Zcol)

        # Plot clusters with ellipses
        for k in range(K):
            mean = np.mean(YA_UASE[0, Z == k], axis=0)
            cov = np.cov(YA_UASE[0, Z == k].T)
            ellipse_x, ellipse_y = gaussian_ellipse(mean, cov)
            axs[i].plot(ellipse_x, ellipse_y, '--', color='black')

            # Plot the mean point in the center of the cluster
            axs[i].scatter(mean[0], mean[1], marker='o', s=20, c='black')

        # Enable grid lines
        axs[i].grid(True)

        axs[i].set_title(f'n = {n}')
        axs[i].set_aspect('equal')

    fig.supylabel("UASE")
    plt.tight_layout()
    plt.show()


# Define initial connectivity matrix B1 for the embedding
B1 = np.array([
    [0.30, 0.15, 0.10],
    [0.15, 0.35, 0.05],
    [0.10, 0.05, 0.20]
])

# Specify the number of communities and the node counts to plot
K = 3
node_counts = [1000, 2000, 4000]

# Plot each scenario with the specified node counts
plot_time_step_with_nodes(B1, K, node_counts)
